Ví dụ OIADB

Trang này cung cấp các ví dụ thực tế về cách sử dụng thư viện OIADB trong các kịch bản khác nhau, từ cơ bản đến nâng cao. Mỗi ví dụ đều có mã nguồn đầy đủ và giải thích chi tiết.

Ví dụ cơ bản

Kết nối thiết bị

Ví dụ này minh họa cách kết nối với thiết bị Android và lấy thông tin cơ bản.

from oiadb import MyADB

def main():
    # Khởi tạo với thiết bị mặc định
    adb = MyADB()
    
    # Liệt kê các thiết bị đã kết nối
    devices = adb.get_devices()
    print(f"Các thiết bị đã kết nối: {devices}")
    
    # Lấy thông tin thiết bị
    device_info = adb.get_device_info()
    print("\nThông tin thiết bị:")
    print(f"Model: {device_info.get('model', 'N/A')}")
    print(f"Manufacturer: {device_info.get('manufacturer', 'N/A')}")
    print(f"Android version: {device_info.get('android_version', 'N/A')}")
    print(f"SDK version: {device_info.get('sdk_version', 'N/A')}")
    print(f"Serial: {device_info.get('serial', 'N/A')}")
    
    # Lấy kích thước màn hình
    width, height = adb.get_screen_size()
    print(f"\nKích thước màn hình: {width}x{height}")
    
    # Kiểm tra trạng thái màn hình
    screen_state = "Bật" if adb.is_screen_on() else "Tắt"
    print(f"Trạng thái màn hình: {screen_state}")

if __name__ == "__main__":
    main()

Tương tác cơ bản

Ví dụ này minh họa các thao tác tương tác cơ bản với thiết bị Android.

from oiadb import MyADB
from oiadb.commands import interaction
import time

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Mở trình khởi chạy ứng dụng
    interaction.home()
    time.sleep(1)
    
    # Vuốt để mở ngăn thông báo
    width, height = adb.get_screen_size()
    interaction.swipe(width // 2, 0, width // 2, height // 2)
    print("Đã mở ngăn thông báo")
    time.sleep(2)
    
    # Đóng ngăn thông báo
    interaction.back()
    print("Đã đóng ngăn thông báo")
    time.sleep(1)
    
    # Mở ứng dụng gần đây
    interaction.recent_apps()
    print("Đã mở ứng dụng gần đây")
    time.sleep(2)
    
    # Vuốt để đóng ứng dụng đầu tiên
    interaction.swipe(width // 2, height // 2, width // 2, 0)
    print("Đã vuốt để đóng ứng dụng")
    time.sleep(1)
    
    # Quay lại màn hình chính
    interaction.home()
    print("Đã quay lại màn hình chính")

if __name__ == "__main__":
    main()

Quản lý ứng dụng

Ví dụ này minh họa cách cài đặt, khởi động và gỡ cài đặt ứng dụng.

from oiadb import MyADB
from oiadb.commands import apps
import time

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Đường dẫn đến file APK (thay thế bằng đường dẫn thực tế)
    apk_path = "/path/to/example.apk"
    package_name = "com.example.app"
    
    # Cài đặt ứng dụng
    print(f"Đang cài đặt {apk_path}...")
    result = apps.install_app(apk_path)
    if result:
        print("Cài đặt thành công")
    else:
        print("Cài đặt thất bại")
        return
    
    # Lấy danh sách ứng dụng đã cài đặt
    installed_apps = apps.get_installed_apps()
    print(f"\nSố lượng ứng dụng đã cài đặt: {len(installed_apps)}")
    
    # Kiểm tra xem ứng dụng đã được cài đặt chưa
    if package_name in installed_apps:
        print(f"{package_name} đã được cài đặt")
    
    # Khởi động ứng dụng
    print(f"\nĐang khởi động {package_name}...")
    apps.start_app(package_name)
    print("Ứng dụng đã được khởi động")
    time.sleep(3)
    
    # Lấy danh sách ứng dụng đang chạy
    running_apps = apps.get_running_apps()
    print(f"\nSố lượng ứng dụng đang chạy: {len(running_apps)}")
    
    # Dừng ứng dụng
    print(f"\nĐang dừng {package_name}...")
    apps.stop_app(package_name)
    print("Ứng dụng đã được dừng")
    time.sleep(1)
    
    # Xóa dữ liệu ứng dụng
    print(f"\nĐang xóa dữ liệu của {package_name}...")
    apps.clear_app_data(package_name)
    print("Dữ liệu ứng dụng đã được xóa")
    
    # Gỡ cài đặt ứng dụng
    print(f"\nĐang gỡ cài đặt {package_name}...")
    result = apps.uninstall_app(package_name)
    if result:
        print("Gỡ cài đặt thành công")
    else:
        print("Gỡ cài đặt thất bại")

if __name__ == "__main__":
    main()

Tìm và nhấp hình ảnh

Ví dụ này minh họa cách tìm và nhấp vào hình ảnh trên màn hình.

from oiadb import MyADB
from oiadb.commands import image_interaction
import time

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Đường dẫn đến hình ảnh mẫu
    settings_icon = "templates/settings_icon.png"
    wifi_button = "templates/wifi_button.png"
    back_button = "templates/back_button.png"
    
    # Mở màn hình chính
    adb.execute_command("shell input keyevent KEYCODE_HOME")
    time.sleep(1)
    
    # Tìm và nhấp vào biểu tượng Cài đặt
    print("Đang tìm biểu tượng Cài đặt...")
    result = image_interaction.tap_image(settings_icon, threshold=0.8)
    if result:
        print("Đã nhấp vào biểu tượng Cài đặt")
    else:
        print("Không tìm thấy biểu tượng Cài đặt")
        return
    
    time.sleep(2)
    
    # Tìm và nhấp vào nút Wi-Fi
    print("\nĐang tìm nút Wi-Fi...")
    result = image_interaction.tap_image(wifi_button, threshold=0.8)
    if result:
        print("Đã nhấp vào nút Wi-Fi")
    else:
        print("Không tìm thấy nút Wi-Fi")
    
    time.sleep(2)
    
    # Tìm và nhấp vào nút Quay lại
    print("\nĐang tìm nút Quay lại...")
    result = image_interaction.tap_image(back_button, threshold=0.8)
    if result:
        print("Đã nhấp vào nút Quay lại")
    else:
        print("Không tìm thấy nút Quay lại")
    
    time.sleep(1)
    
    # Quay lại màn hình chính
    adb.execute_command("shell input keyevent KEYCODE_HOME")
    print("\nĐã quay lại màn hình chính")

if __name__ == "__main__":
    main()

Đợi và nhấp hình ảnh

Ví dụ này minh họa cách đợi hình ảnh xuất hiện trên màn hình và nhấp vào nó.

from oiadb import MyADB
from oiadb.commands import image_interaction, apps
import time

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Đường dẫn đến hình ảnh mẫu
    app_icon = "templates/app_icon.png"
    accept_button = "templates/accept_button.png"
    skip_button = "templates/skip_button.png"
    
    # Mở màn hình chính
    adb.execute_command("shell input keyevent KEYCODE_HOME")
    time.sleep(1)
    
    # Tìm và nhấp vào biểu tượng ứng dụng
    print("Đang tìm biểu tượng ứng dụng...")
    result = image_interaction.tap_image(app_icon, threshold=0.8)
    if not result:
        print("Không tìm thấy biểu tượng ứng dụng")
        return
    
    print("Đã nhấp vào biểu tượng ứng dụng")
    
    # Đợi và nhấp vào nút Chấp nhận (tối đa 10 giây)
    print("\nĐang đợi nút Chấp nhận xuất hiện (tối đa 10 giây)...")
    result = image_interaction.wait_and_tap_image(
        accept_button,
        timeout=10,      # Thời gian tối đa đợi (giây)
        interval=0.5,    # Khoảng thời gian giữa các lần tìm kiếm (giây)
        threshold=0.8    # Ngưỡng tương đồng
    )
    
    if result:
        print("Đã nhấp vào nút Chấp nhận")
    else:
        print("Không tìm thấy nút Chấp nhận trong thời gian chờ")
    
    # Đợi và nhấp vào nút Bỏ qua (tối đa 5 giây)
    print("\nĐang đợi nút Bỏ qua xuất hiện (tối đa 5 giây)...")
    result = image_interaction.wait_and_tap_image(
        skip_button,
        timeout=5,
        interval=0.5,
        threshold=0.8
    )
    
    if result:
        print("Đã nhấp vào nút Bỏ qua")
    else:
        print("Không tìm thấy nút Bỏ qua trong thời gian chờ")
    
    # Đợi 3 giây để xem kết quả
    time.sleep(3)
    
    # Dừng ứng dụng
    apps.stop_app("com.example.app")
    print("\nĐã dừng ứng dụng")

if __name__ == "__main__":
    main()

Tự động chơi game

Ví dụ này minh họa cách sử dụng nhận diện hình ảnh để tự động chơi một trò chơi đơn giản.

from oiadb import MyADB
from oiadb.commands import image_interaction, apps
import time
import random

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Đường dẫn đến hình ảnh mẫu
    game_icon = "templates/game_icon.png"
    play_button = "templates/play_button.png"
    target_icon = "templates/target_icon.png"
    retry_button = "templates/retry_button.png"
    
    # Mở màn hình chính
    adb.execute_command("shell input keyevent KEYCODE_HOME")
    time.sleep(1)
    
    # Khởi động trò chơi
    print("Đang khởi động trò chơi...")
    result = image_interaction.tap_image(game_icon, threshold=0.8)
    if not result:
        print("Không tìm thấy biểu tượng trò chơi")
        return
    
    # Đợi và nhấp vào nút Chơi
    print("\nĐang đợi nút Chơi xuất hiện...")
    result = image_interaction.wait_and_tap_image(
        play_button,
        timeout=10,
        interval=0.5,
        threshold=0.8
    )
    
    if not result:
        print("Không tìm thấy nút Chơi")
        return
    
    print("Đã bắt đầu trò chơi")
    
    # Chơi trong 60 giây
    print("\nĐang chơi trò chơi trong 60 giây...")
    start_time = time.time()
    targets_clicked = 0
    
    while time.time() - start_time < 60:
        # Tìm tất cả các mục tiêu trên màn hình
        results = image_interaction.find_all_images(
            target_icon,
            threshold=0.8,
            max_results=5  # Tìm tối đa 5 mục tiêu cùng một lúc
        )
        
        if results:
            # Nhấp vào mục tiêu ngẫu nhiên
            target = random.choice(results)
            x, y, confidence = target
            image_interaction.tap(x, y)
            targets_clicked += 1
            print(f"Đã nhấp vào mục tiêu #{targets_clicked} tại ({x}, {y})")
            time.sleep(0.2)  # Đợi một chút giữa các lần nhấp
        else:
            # Kiểm tra xem trò chơi có kết thúc không
            if image_interaction.is_image_present(retry_button, threshold=0.8):
                print("\nTrò chơi đã kết thúc, đang bắt đầu lại...")
                image_interaction.tap_image(retry_button)
                time.sleep(1)
            
            time.sleep(0.1)  # Đợi một chút trước khi tìm kiếm lại
    
    print(f"\nĐã kết thúc. Tổng số mục tiêu đã nhấp: {targets_clicked}")
    
    # Dừng trò chơi
    apps.stop_app("com.example.game")
    print("Đã dừng trò chơi")

if __name__ == "__main__":
    main()

XML Dump

Điền form tự động

Ví dụ này minh họa cách sử dụng XML Dump để điền form tự động.

from oiadb import MyADB
from oiadb.commands import xml_dump, apps
import time

def main():
    # Khởi tạo ADB
    adb = MyADB()
    
    # Khởi động ứng dụng
    print("Đang khởi động ứng dụng...")
    apps.start_app("com.example.form")
    time.sleep(2)
    
    # Lấy cấu trúc XML của giao diện người dùng
    print("\nĐang lấy cấu trúc UI...")
    xml_tree = xml_dump.get_ui_xml()
    
    # Tìm và điền vào trường Tên
    print("\nĐang điền form...")
    name_field = xml_dump.find_element_by_attribute("hint", "Tên")
    if name_field:
        xml_dump.click_element(name_field)
        xml_dump.input_text(name_field, "Nguyễn Văn A")
        print("Đã điền trường Tên")
    else:
        print("Không tìm thấy trường Tên")
    
    # Cập nhật cấu trúc XML sau khi tương tác
    xml_tree = xml_dump.get_ui_xml()
    
    # Tìm và điền vào trường Email
    email_field = xml_dump.find_element_by_attribute("hint", "Email")
    if email_field:
        xml_dump.click_element(email_field)
        xml_dump.input_text(email_field, "nguyenvana@example.com")
        print("Đã điền trường Email")
    else:
        print("Không tìm thấy trường Email")
    
    # Cập nhật cấu trúc XML sau khi tương tác
    xml_tree = xml_dump.get_ui_xml()
    
    # Tìm và điền vào trường Số điện thoại
    phone_field = xml_dump.find_element_by_attribute("hint", "Số điện thoại")
    if phone_field:
        xml_dump.click_element(phone_field)
        xml_dump.input_text(phone_field, "0123456789")
        print("Đã điền trường Số điện thoại")
    else:
        print("Không tìm thấy trường Số điện thoại")
    
    # Cập nhật cấu trúc XML sau khi tương tác
    xml_tree = xml_dump.get_ui_xml()
    
    # Tìm và nhấp vào nút Gửi
    submit_button = xml_dump.find_element_by_text("Gửi")
    if submit_button:
        xml_dump.click_element(submit_button)
        print("Đã nhấp vào nút Gửi")
    else:
        print("Không tìm thấy nút Gửi")
    
    # Đợi thông báo xác nhận
    time.sleep(2)
    xml_tree = xml_dump.get_ui_xml()
    
    # Kiểm tra xem form đã được gửi thành công chưa
    success_message = xml_dump.find_element_by_text("Gửi thành công")
    if success_message:
        print("\nForm đã được gửi thành công!")
    else:
        print("\nKhông tìm thấy thông báo xác nhận")
    
    # Dừng ứng dụng
    time.sleep(2)
    apps.stop_app("com.example.form")
    print("Đã dừng ứng dụng")

if __name__ == "__main__":
    main()
